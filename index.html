// TITOLO DINAMICO

function aggiornaTitolo() {

  const meseSel = document.getElementById("mese").value;

  const annoSel = document.getElementById("anno").value;


  let testo = "Acquisti";


  if (annoSel !== "all" && meseSel === "all") testo += ` (${annoSel})`;

  else if (annoSel !== "all" && meseSel !== "all") testo += ` (${annoSel} / ${meseSel})`;

  else if (annoSel === "all" && meseSel !== "all") testo += ` (Mese ${meseSel})`;


  document.getElementById("titolo").innerText = testo;

}


// AGGIORNA LISTA

function aggiorna() {

  const meseSel = document.getElementById("mese").value;

  const annoSel = document.getElementById("anno").value;


  db.collection("spese").orderBy("timestamp").onSnapshot(snapshot => {

    lista.innerHTML = "";

    let totale = 0;


    snapshot.forEach(doc => {

      const s = doc.data();

      const data = new Date(s.timestamp);

      const m = data.getMonth() + 1;

      const a = data.getFullYear();


      if (annoSel !== "all" && a != annoSel) return;

      if (meseSel !== "all" && m != meseSel) return;


      totale += s.importo;


      lista.innerHTML += `

        <div class="item">

          <div>${s.motivazione}</div>

          <div>€ ${s.importo.toFixed(2)}</div>

          <button class="remove" onclick="rimuovi('${doc.id}')">X</button>

        </div>`;

    });


    totale.innerText = "Totale: € " + totale.toFixed(2);

    aggiornaTitolo();

  });

}


// AGGIUNGI

function aggiungi() {

  if (!motivazione.value || !importo.value) return;


  db.collection("spese").add({

    motivazione: motivazione.value,

    importo: parseFloat(importo.value),

    timestamp: Date.now()

  });


  motivazione.value = "";

  importo.value = "";

}


// RIMUOVI

function rimuovi(id) {

}


// EXPORT EXCEL

function esportaExcel() {

  const meseSel = document.getElementById("mese").value;

  const annoSel = document.getElementById("anno").value;


  db.collection("spese").orderBy("timestamp").get().then(snapshot => {

    const dati = [];

    let totale = 0;


    snapshot.forEach(doc => {

      const s = doc.data();

      const data = new Date(s.timestamp);

      const m = data.getMonth() + 1;

      const a = data.getFullYear();


      if (annoSel !== "all" && a != annoSel) return;

      if (meseSel !== "all" && m != meseSel) return;


      totale += s.importo;


      dati.push({

        "Motivazione": s.motivazione,

        "Importo (€)": s.importo,

        "Data": data.toLocaleString()

      });

    });


    dati.push({ "Motivazione": "TOTALE", "Importo (€)": totale, "Data": "" });


    const ws = XLSX.utils.json_to_sheet(dati);

    const wb = XLSX.utils.book_new();

    XLSX.utils.book_append_sheet(wb, ws, "Spese");


    XLSX.writeFile(wb, "spese.xlsx");

  });

}


// BACKUP AUTOMATICO

function creaBackup() {

  db.collection("spese").get().then(snapshot => {

    const dati = [];

    snapshot.forEach(doc => dati.push(doc.data()));


    db.collection("backup").add({

      timestamp: Date.now(),

      dati: dati

    }).then(() => alert("Backup creato"));

  });

}


// RIPRISTINO

function ripristinaUltimoBackup() {

  db.collection("backup").orderBy("timestamp", "desc").limit(1).get().then(snapshot => {

    if (snapshot.empty) return alert("Nessun backup");


    const backup = snapshot.docs[0].data().dati;


    db.collection("spese").get().then(speseSnap => {

      const batch = db.batch();

      speseSnap.forEach(doc => batch.delete(doc.ref));

      batch.commit().then(() => {

        backup.forEach(s => db.collection("spese").add(s));

      });

    });

  });

}


// BACKUP JSON

function esportaBackupJSON() {

  db.collection("spese").get().then(snapshot => {

    const dati = [];

    snapshot.forEach(doc => dati.push(doc.data()));


    const blob = new Blob([JSON.stringify(dati, null, 2)], { type: "application/json" });

    const url = URL.createObjectURL(blob);


    const a = document.createElement("a");

    a.href = url;

    a.download = "backup_spese.json";

    document.body.appendChild(a);

    a.click();

    document.body.removeChild(a);

    URL.revokeObjectURL(url);

  });

}
